若n为表长，n = 0时为空表。

## 1. 顺序表
### 1.1 理论
- 定义：把逻辑上相邻的数据元素存储在物理上相邻的存储单元中的顺序存储结构。
线性表的顺序表示又称为顺序存储结构或顺序映像，简称顺序表。
第1个数据元素存储位置成为线性表的起始地址或基地址。

- 物理存储要求：依次存储，地址连续，**中间没有空出存储单元**，占用一片连续的存储空间。
- 顺序表特点：
	1. 以物理位置相邻表示逻辑关系
	2. 任一元素可随机存取（优点）

**顺序表与数组**
- 共同点：地址连续、依次存放、随寄存取、类型相同
- 差别：线性表长度可变（删除），数组长度不可动态定义

**多项式的顺序存储结构定义**

$$
P_n(x) = p_1x^{e_1} + p_2x^{e_2} + ... + p_mx^{e_m}.
$$

线性表 $P((p_1, e_1), (p_2, e_2), ... , (p_m, e_m))$
### 1.2 实现
#### 1.2.1 预定义
~~~cpp
#define   TRUE 1
#define   FALSE 0
#define   OK 1
#define   ERROR 0
#define   INFEASIBLE -1
#define   SQ_OVERFLOW -2
~~~
#### 1.2.2 顺序表类定义
~~~cpp
class SqList {
public:
	//Status  是函数的类型，其值是函数结果状态代码
	typedef   int   Status;
	typedef   char   ElemType;
	SqList();
	~SqList() = default;
	void Destroy();
	void Clear();
	size_t GetLength() const;
	size_t GetSize() const;
	Status IsEmpty() const;
	Status GetElem(size_t i, ElemType& e) const;
	size_t LocateElem(const ElemType& e) const;
	Status Insert_Sq(size_t i, const ElemType& e);
	Status Delete(size_t i, ElemType& e);
private:
	void reallocate();
	shared_ptr<ElemType> elem;
	size_t length;
	size_t size;
};
~~~
#### 1.2.2 顺序表构造函数
~~~cpp
SqList::SqList() : elem(new ElemType[1]), length(0), size(1) {
	if (!elem)
		exit(SQ_OVERFLOW);
}
~~~

#### 1.2.8 获取第i个元素
~~~cpp
SqList::Status SqList::GetElem(size_t i, ElemType& e) const
{
	if (i < 1 || i > length)
		return ERROR;
	e = elem.get()[i - 1];
	return OK;
}
~~~
#### 1.2.9 定位元素
~~~cpp
size_t SqList::LocateElem(const ElemType& e) const
{
	for (size_t i = 0; i < length; ++i)
		if (elem.get()[i] == e)
			return i + 1;
	return 0;
}
~~~
#### 1.2.10 重新分配空间
~~~cpp
void SqList::reallocate()
{
	auto newspace = size ? size * 2 : 1;
	shared_ptr<ElemType> temp(new ElemType[newspace]);
	for (size_t i = 0; i < length; ++i)
		temp.get()[i] = std::move(elem.get()[i]);
	elem = temp;
	size = newspace;
}
~~~
#### 1.2.11 插入元素
~~~cpp
SqList::Status SqList::Insert_Sq(size_t i, const ElemType& e)
{
	if (i < 1 || i > length + 1)
		return ERROR;
	if (length == size)
		reallocate();
	for (size_t j = length; j >= i; --j)
		elem.get()[j] = elem.get()[j - 1];
	elem.get()[i - 1] = e;
	length++;	
	return OK;
}
~~~
#### 1.2.12 删除元素
~~~cpp
SqList::Status SqList::Delete(size_t i, ElemType& e)
{
	if (i < 1 || i > length)
		return ERROR;
	e = elem.get()[i - 1];
	for (size_t j = i - 1; j < length; ++j)
		elem.get()[j] = elem.get()[j + 1];
	length--;
	return OK;
}
~~~
## 2 单链表
### 2.1 理论
- 定义：线性表的链式表示又称为非顺序映像或链式映像
- 用一组物理位置任意的存储单元来存放线性表的数据元素
- 结点由数据域和指针域构成，数据域存储数据元素，指针域存放的元素称为链或指针，指向下一个结点的地址，第一个结点的地址为头指针，最后一个结点的指针设置成NULL

**分类**

- 单链表：结点只有一个指针域，只指向后继
- 双链表：结点有两个指针域，分别指向前驱和后继
- 循环链表：首位相接

**三个概念**

- 头指针：指向链表第一个结点的指针
- 首元结点：链表中存储第一个数据元素 $a_1$ 的指针
- 头结点：在链表首元结点之前附设的一个结点

无头结点时，头指针为空时表示空表

有头结点时，头结点指针域为空时表示空表

**设置头结点好处**

1. 首元结点处理与其他位置相同
2. 便于空表与非空表的统一处理	

**顺序表是随机存取法，链表是顺序存取法，顺序表查找第一个元素和最后一个元素耗时相同，链表不同**

单链表由表头唯一确定，因此单链表可以用头指针的名字命名，若头指针名L，则称链表为表L。

单链表的存储结构

~~~c
typedef struct Lnode{
	ElemType data;
	struct Lnode* next;
}LNode, *LinkList;
~~~

定义链表L: LinkList L;

定义结点指针p: LNode* p;

### 2.2 实现
#### 2.2.1 单链表初始化
~~~c
Status InitList_L(LinkList &L)
{
	L = new LNode;
	//或(LinkList)malloc(sizeof(LNode))
	L->next = NULL;
	return OK;
}
~~~
#### 2.2.2 判断链表为空
~~~c
Status ListEmpty(const LinkList& L)
{
	if (L->next)
		return FALSE;
	else
		return TRUE;
}
~~~
#### 2.2.3 销毁链表
~~~c
Status DestroyList(LinkList& L)
{
	LNode* p;
	while (L)
	{
		p = L;
		L = L->next;
		delete p;
	}
	return OK;
}
~~~
#### 2.2.4 清除链表
~~~c
Status ClearList(LinkList& L)
{
	LNode* p, * q;
	p = L->next;
	while (p)
	{
		q = p->next;
		delete p;
		p = q;
	}
	L->next = NULL;
	return OK;
}
~~~
#### 2.2.5 获取链表长度
~~~c
int ListLength(const LinkList& L)
{
	LNode* p;
	int i = 0;
	p = L->next;
	while (p)
	{
		++i;
		p = p->next;
	}
	return i;
}
~~~
#### 2.2.6 获取链表第i个元素
~~~c
Status GetElem_L(const LinkList& L, int i, ElemType& e)
{
	int j = 1;
	LNode* p = L->next;
	while (p && j < i)
	{
		p = p->next;
		++j;
	}
	if (!p || j > i)
		return ERROR;
	e = p->data;
	return OK;
}
~~~
#### 2.2.7 根据值定位元素（返回地址）
~~~c
LNode* LocateElem_L(LinkList& L, const ElemType& e)
{
	LNode* p = L->next;
	while (p && p->data != e)
	{
		p = p->next;
	}
	return p;
}
~~~
#### 2.2.8 根据值定位元素（返回位置）
~~~c
int LocateElem_L_i(LinkList& L, const ElemType& e)
{
	int j = 1;
	LNode* p = L->next;
	while (p && p->data != e)
	{
		p = p->next;
		++j;
	}
	if (!p)
		return ERROR;
	else
		return j;
}
~~~
#### 2.2.9 在第i个位置插入元素
~~~c
Status ListInsert_L(LinkList& L, int i, const ElemType& e)
{
	int j = 0;
	LNode* p = L;
	//这里可以在尾结点后面插入，也就是让尾结点成为第i - 1个结点
	while (p && j < i - 1)
	{
		p = p->next;
		++j;
	}
	if (!p || j > i - 1) return ERROR;
	LNode* q = new LNode;
	q->data = e;
	q->next = p->next;
	p->next = q;
	return OK;
}
~~~
#### 2.2.10 删除第i个位置元素
~~~c
Status DeleteList_L(LinkList& L, int i, ElemType& e)
{
	int j = 0;
	LNode* p = L;
	//这里要注意要确保被删除的结点必须存在，也就是地址不为空
	while (p->next && j < i - 1)
	{
		p = p->next;
		++j;
	}
	if (!(p->next) || j > i - 1) return ERROR;
	LNode* q = p->next;
	p->next = q->next;
	e = q->data;
	delete q;
	return OK;
}
~~~
#### 2.2.11 创建n个结点的链表（头插法）
~~~c
void CreatList_H(LinkList& L, int n)
{
	L = new LNode;
	L->next = NULL;
	for (int i = 0; i < n; ++i)
	{
		LNode* p = new LNode;
		cin >> p->data;
		p->next = L->next;
		L->next = p;
	}
}
~~~
#### 2.2.12 创建n个结点的链表（尾插法）
~~~c
void CreatList_T(LinkList& L, int n)
{
	L = new LNode;
	L->next = NULL;
	LNode* p = L;
	for (int i = 0; i < n; ++i)
	{
		LNode* q = new LNode;
		cin >> q->data;
		q->next = NULL;
		p->next = q;
		p = q;
	}
}
~~~
